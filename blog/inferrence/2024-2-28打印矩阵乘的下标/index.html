c = list()
for i in range(N):
    for j in range(M):
        c.append([])
        for kk in range(K):
            c[i*M+j].append('a['+str(i)+','+str(kk)+']*b['+str(kk)+','+str(j)+']'+'+'if kk<K-1 else 'a['+str(i)+','+str(kk)+']*b['+str(kk)+','+str(j)+']')

for i in range(N):
    for j in range(M):
        for k in range(K):
            print(c[i*M+j][k],end='')
        print('    ',end= '')
    print('')
